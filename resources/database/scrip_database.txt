-- 1. QUẢN LÝ NGƯỜI DÙNG

CREATE TABLE Users (
UserID INT PRIMARY KEY IDENTITY(1,1),
UserName NVARCHAR(50) NOT NULL UNIQUE,
PassWord NVARCHAR(255) NOT NULL,
Email NVARCHAR(100) NOT NULL UNIQUE,
Phone NVARCHAR(20),
FullName NVARCHAR(100),
Role NVARCHAR(50) NOT NULL DEFAULT 'customer',
Status NVARCHAR(20) NOT NULL DEFAULT 'active',
CreatedAt DATETIME DEFAULT GETDATE(),
LastLogin DATETIME,
Avatar NVARCHAR(255),
IsDeleted BIT DEFAULT 0,
Reset_Token NVARCHAR(10) NULL,
Reset_Token_Expiry DATETIME NULL,
Google_Id NVARCHAR(100) NULL
);

CREATE TABLE Addresses (
AddressID INT PRIMARY KEY IDENTITY(1,1),
UserID INT NOT NULL,
RecipientName NVARCHAR(100),
Phone NVARCHAR(20),
Province NVARCHAR(50),
District NVARCHAR(50),
Ward NVARCHAR(50),
AddressDetail NVARCHAR(255),
IsDefault BIT DEFAULT 0,
AddressType NVARCHAR(20) CHECK (AddressType IN ('shipping', 'billing')) DEFAULT 'shipping',
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE
);

-- 2. QUẢN LÝ SẢN PHẨM CHĂM SÓC DA

CREATE TABLE Categories (
CategoryID INT PRIMARY KEY IDENTITY(1,1),
ParentCategoryID INT,
CategoryName NVARCHAR(100),
Description NVARCHAR(MAX),
Image NVARCHAR(255),
Status NVARCHAR(20) CHECK (Status IN ('active', 'inactive')) DEFAULT 'active',
DisplayOrder INT,
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (ParentCategoryID) REFERENCES Categories(CategoryID) ON DELETE NO ACTION ON UPDATE NO ACTION
);

CREATE TABLE Products (
ProductID INT PRIMARY KEY IDENTITY(1,1),
CategoryID INT,
ProductName NVARCHAR(255) NOT NULL,
BrandName NVARCHAR(100) NOT NULL,
ShortDescription NVARCHAR(255),
Price DECIMAL(12,2) NOT NULL,
SalePrice DECIMAL(12,2),
Quantity INT NOT NULL,
SKU NVARCHAR(50) UNIQUE,
QuantityValue DECIMAL(8,2),
QuantityUnit NVARCHAR(20),
Status NVARCHAR(20) CHECK (Status IN ('active', 'inactive', 'out_of_stock')) DEFAULT 'active',
Featured BIT DEFAULT 0,
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (CategoryID) REFERENCES Categories(CategoryID) ON DELETE SET NULL
);

CREATE TABLE ProductDetails (
ProductDetailID INT PRIMARY KEY IDENTITY(1,1),
ProductID INT NOT NULL,
Description NVARCHAR(MAX),
Ingredients NVARCHAR(MAX),
Origin NVARCHAR(50),
SkinConcerns NVARCHAR(255),
StorageInstructions NVARCHAR(255),
HowToUse NVARCHAR(MAX),
ManufactureDate DATE,
ExpiryDate DATE,
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE
);

CREATE TABLE ProductImages (
ImageID INT PRIMARY KEY IDENTITY(1,1),
ProductID INT NOT NULL,
ImageURL NVARCHAR(255) NOT NULL,
IsMainImage BIT DEFAULT 0,
DisplayOrder INT,
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE
);

-- 3. QUẢN LÝ GIỎ HÀNG

CREATE TABLE Carts (
CartID INT PRIMARY KEY IDENTITY(1,1),
UserID INT,
SessionID VARCHAR(255),
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE
);

CREATE TABLE CartItems (
CartItemID INT PRIMARY KEY IDENTITY(1,1),
CartID INT,
ProductID INT,
Quantity INT,
AddedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (CartID) REFERENCES Carts(CartID) ON DELETE CASCADE,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE NO ACTION ON UPDATE NO ACTION
);

-- 4. QUẢN LÝ ĐƠN HÀNG VÀ THANH TOÁN

CREATE TABLE PaymentMethods (
PaymentMethodID INT PRIMARY KEY IDENTITY(1,1),
MethodName NVARCHAR(100) NOT NULL,
Description NVARCHAR(500),
IsActive BIT DEFAULT 1,
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0
);

CREATE TABLE Orders (
OrderID INT PRIMARY KEY IDENTITY(1,1),
UserID INT,
OrderNumber NVARCHAR(50) UNIQUE,
Status NVARCHAR(20) CHECK (Status IN ('pending', 'confirmed', 'processing', 'shipping', 'delivered', 'cancelled', 'refunded')) DEFAULT 'pending',
TotalAmount DECIMAL(12,2),
DiscountAmount DECIMAL(12,2),
ShippingFee DECIMAL(12,2),
Tax DECIMAL(12,2),
FinalAmount DECIMAL(12,2),
PaymentMethodID INT,
ShippingAddressID INT,
BillingAddressID INT,
Notes NVARCHAR(MAX),
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE SET NULL,
FOREIGN KEY (PaymentMethodID) REFERENCES PaymentMethods(PaymentMethodID) ON DELETE SET NULL,
FOREIGN KEY (ShippingAddressID) REFERENCES Addresses(AddressID) ON DELETE NO ACTION ON UPDATE NO ACTION,
FOREIGN KEY (BillingAddressID) REFERENCES Addresses(AddressID) ON DELETE NO ACTION ON UPDATE NO ACTION
);

CREATE TABLE OrderItems (
OrderItemID INT PRIMARY KEY IDENTITY(1,1),
OrderID INT,
ProductID INT,
ProductName NVARCHAR(255),
Quantity INT,
UnitPrice DECIMAL(12,2),
Subtotal DECIMAL(12,2),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (OrderID) REFERENCES Orders(OrderID) ON DELETE CASCADE,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE NO ACTION ON UPDATE NO ACTION
);

CREATE TABLE Payments (
PaymentID INT PRIMARY KEY IDENTITY(1,1),
OrderID INT,
PaymentMethodID INT,
TransactionID NVARCHAR(100),
Amount DECIMAL(12,2),
Status NVARCHAR(20) CHECK (Status IN ('pending', 'completed', 'failed', 'refunded')) DEFAULT 'pending',
PaymentDate DATETIME,
PaymentDetails NVARCHAR(MAX),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (OrderID) REFERENCES Orders(OrderID) ON DELETE CASCADE,
FOREIGN KEY (PaymentMethodID) REFERENCES PaymentMethods(PaymentMethodID) ON DELETE SET NULL
);

-- 5. KHUYẾN MÃI VÀ MÃ GIẢM GIÁ

CREATE TABLE Coupons (
CouponID INT PRIMARY KEY IDENTITY(1,1),
Code NVARCHAR(50) UNIQUE,
Description NVARCHAR(MAX),
DiscountType NVARCHAR(20) CHECK (DiscountType IN ('percentage', 'fixed_amount')),
DiscountValue DECIMAL(12,2),
MinimumOrder DECIMAL(12,2),
MaximumDiscount DECIMAL(12,2),
UsageLimit INT,
UsageCount INT DEFAULT 0,
StartDate DATETIME,
EndDate DATETIME,
Status NVARCHAR(20) CHECK (Status IN ('active', 'inactive', 'expired')) DEFAULT 'active',
CreatedAt DATETIME DEFAULT GETDATE(),
UpdatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0
);

CREATE TABLE CouponProducts (
CouponProductID INT PRIMARY KEY IDENTITY(1,1),
CouponID INT,
ProductID INT,
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (CouponID) REFERENCES Coupons(CouponID) ON DELETE CASCADE,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE
);

CREATE TABLE CouponUsage (
UsageID INT PRIMARY KEY IDENTITY(1,1),
CouponID INT,
UserID INT,
OrderID INT,
DiscountAmount DECIMAL(12,2),
UsedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (CouponID) REFERENCES Coupons(CouponID) ON DELETE CASCADE,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
FOREIGN KEY (OrderID) REFERENCES Orders(OrderID) ON DELETE CASCADE
);

CREATE TABLE InventoryLogs (
LogID INT PRIMARY KEY IDENTITY(1,1),
ProductID INT,
QuantityBefore INT,
QuantityAfter INT,
ChangeType NVARCHAR(20) CHECK (ChangeType IN ('increase', 'decrease')),
Reason NVARCHAR(255),
ReferenceID INT,
ReferenceType NVARCHAR(50),
ExpiryDate DATE,
CreatedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE
);

-- 6. TƯƠNG TÁC KHÁCH HÀNG

CREATE TABLE Reviews (
ReviewID INT PRIMARY KEY IDENTITY(1,1),
ProductID INT,
UserID INT,
OrderID INT,
Rating INT CHECK (Rating BETWEEN 1 AND 5),
Comment NVARCHAR(MAX),
SkinCondition NVARCHAR(255),
ReviewDate DATETIME DEFAULT GETDATE(),
Status NVARCHAR(20) CHECK (Status IN ('pending', 'approved', 'rejected')) DEFAULT 'pending',
IsVerifiedPurchase BIT DEFAULT 0, 
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
FOREIGN KEY (OrderID) REFERENCES Orders(OrderID) ON DELETE SET NULL
);

CREATE TABLE Wishlist (
WishlistID INT PRIMARY KEY IDENTITY(1,1),
UserID INT,
ProductID INT,
AddedAt DATETIME DEFAULT GETDATE(),
IsDeleted BIT DEFAULT 0,
FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
FOREIGN KEY (ProductID) REFERENCES Products(ProductID) ON DELETE CASCADE
);
